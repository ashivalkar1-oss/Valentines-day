  <script>

    const card = document.getElementById("card");

    const buttons = document.getElementById("buttons");

    const yesBtn = document.getElementById("yesBtn");

    const noBtn = document.getElementById("noBtn");

    const hearts = document.getElementById("hearts");

    let yesScale = 1;

    let noTries = 0;

    // Place "No" near "Yes" initially

    function resetNoPosition() {

      const container = buttons.getBoundingClientRect();

      const yesRect = yesBtn.getBoundingClientRect();

      // position no near the right of yes, but inside container

      const left = (yesRect.left - container.left) + yesRect.width + 14;

      noBtn.style.left = Math.min(left, container.width - noBtn.offsetWidth) + "px";

      noBtn.style.top = "6px";

    }

    // Pick a new random position within the button container

    function moveNoSomewhereElse() {

      const container = buttons.getBoundingClientRect();

      const pad = 6;

      const maxX = Math.max(pad, container.width - noBtn.offsetWidth - pad);

      const maxY = Math.max(pad, container.height - noBtn.offsetHeight - pad);

      // try to keep it away from the yes button area a bit

      const yesRect = yesBtn.getBoundingClientRect();

      const yesCenterX = yesRect.left + yesRect.width / 2 - container.left;

      const yesCenterY = yesRect.top + yesRect.height / 2 - container.top;

      let x, y, tries = 0;

      do {

        x = pad + Math.random() * (maxX - pad);

        y = pad + Math.random() * (maxY - pad);

        tries++;

      } while (tries < 20 && Math.hypot(x - yesCenterX, y - yesCenterY) < 120);

      noBtn.style.left = x + "px";

      noBtn.style.top = y + "px";

    }

    function growYes() {

      noTries++;

      // growth curve: bigger each time, but not insane

      yesScale *= 1.18;

      yesScale = Math.min(yesScale, 6); // cap so it doesn't break the layout

      yesBtn.style.transform = `scale(${yesScale})`;

    }

    function popHearts(count = 10) {

      const bounds = card.getBoundingClientRect();

      for (let i = 0; i < count; i++) {

        const s = document.createElement("div");

        s.className = "heart";

        s.textContent = Math.random() > 0.5 ? "💖" : "💘";

        s.style.left = (20 + Math.random() * (bounds.width - 40)) + "px";

        s.style.top = (bounds.height - 30) + "px";

        s.style.animationDelay = (Math.random() * 0.25) + "s";

        hearts.appendChild(s);

        s.addEventListener("animationend", () => s.remove());

      }

    }

    // Make "No" hard to click:

    // - on mouse move (hover), it runs

    // - on pointer down (attempt click), it runs + yes grows

    noBtn.addEventListener("mouseenter", () => {

      moveNoSomewhereElse();

    });

    noBtn.addEventListener("pointerdown", (e) => {

      e.preventDefault();

      moveNoSomewhereElse();

      growYes();

      popHearts(4);

    });

    // Mobile: if she taps around and hits "No", still run away

    noBtn.addEventListener("click", (e) => {

      e.preventDefault();

      moveNoSomewhereElse();

      growYes();

      popHearts(4);

    });

    yesBtn.addEventListener("click", () => {

      card.classList.add("done");

      popHearts(18);

    });

    // Initial layout

    window.addEventListener("load", resetNoPosition);

    window.addEventListener("resize", resetNoPosition);

  </script>
